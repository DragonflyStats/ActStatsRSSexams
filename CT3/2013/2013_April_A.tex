\begin{eqnarray*}
P [ X > 10 ] &=& 1 − P [ X \leq 10 ] \\ &=& 1 − F ( 10 ) \\ &=& 0.5 \\
\end{eqnarray*}

\begin{eqnarray*}
P [ X < 30 ]  &=& P [ X \leq 20 ] \\ &=& F ( 20 ) \\ &=& 0.7\\
\end{eqnarray*}

\begin{eqnarray*}
P [ X =40 ] &=& F ( 40 ) − F ( 30 ) \\ &=& 0.1\\
\end{eqnarray*}

\begin{eqnarray*}
P [ 20 < X < 50 ]  &=& F ( 40 ) − F ( 20 ) \\ &=& 0.25\\
